# Dynamic

## Anatomy

<FrameworkContent fw="vue">
```vue
<script setup lang="ts">
import { Dynamic } from '@destyler-ui/vue'
</script>

<template>
  <Dynamic.Root>
    <Dynamic.Label />
    <Dynamic.Control>
      <Dynamic.Item>
        <Dynamic.ItemPreview>
          <Dynamic.ItemText />
          <Dynamic.ItemDeleteTrigger />
        </Dynamic.ItemPreview>
        <Dynamic.ItemInput />
      </Dynamic.Item>
      <Dynamic.Input />
      <Dynamic.ClearTrigger />
    </Dynamic.Control>
    <Dynamic.HiddenInput />
  </Dynamic.Root>
</template>
```
